CREATE TABLE `usuario` 
( `id` INT NOT NULL AUTO_INCREMENT , 
  `correo` VARCHAR(50) NOT NULL , 
  `clave` VARCHAR(30) NOT NULL , 
  `nombre` VARCHAR(50) NOT NULL , 
  `perfil` VARCHAR(30) NOT NULL , 
  `agregado` tinyint(1) NOT NULL ,  PRIMARY KEY (`id`)) 
  ENGINE = InnoDB CHARACTER SET latin1 COLLATE latin1_spanish_ci;

CREATE TABLE `producto` 
( `id` INT NOT NULL AUTO_INCREMENT , 
  `nombre` VARCHAR(50) NOT NULL , 
  `descripcion` VARCHAR(250) NOT NULL , 
  `precio` FLOAT NOT NULL , 
  `foto` VARCHAR(250) NOT NULL , 
  `vigente` tinyint(1) NOT NULL ,  PRIMARY KEY (`id`)) 
  ENGINE = InnoDB CHARACTER SET latin1 COLLATE latin1_spanish_ci;

CREATE TABLE `oferta` 
( `id` INT NOT NULL AUTO_INCREMENT , 
  `descripcion` VARCHAR(250) NOT NULL , 
  `precio` FLOAT NOT NULL ,  
  `disponible` tinyint(1) NOT NULL ,
  `fechaInicio` DATETIME NOT NULL ,
  `fechaFin` DATETIME NOT NULL ,  PRIMARY KEY (`id`)) 
  ENGINE = InnoDB CHARACTER SET latin1 COLLATE latin1_spanish_ci;

CREATE TABLE `pedido` 
( `id` INT NOT NULL AUTO_INCREMENT , 
  `descripcion` VARCHAR(250) NOT NULL , 
  `precio` FLOAT NOT NULL , 
  `unidades` INT NOT NULL , 
  `enviado` tinyint(1) NOT NULL ,
  `fechaPedido` DATETIME NOT NULL ,PRIMARY KEY (`id`)) 
  ENGINE = InnoDB CHARACTER SET latin1 COLLATE latin1_spanish_ci;

CREATE TABLE `local` 
( `idLocal` INT NOT NULL AUTO_INCREMENT , 
  `localidad` VARCHAR(250) NOT NULL , 
  `calle` VARCHAR(250) NOT NULL ,
  `numero` INT NOT NULL , 
  `latitud` DECIMAL(65,30) NOT NULL ,
  `longitud` DECIMAL(65,30) NOT NULL , PRIMARY KEY (`idLocal`)) 
  ENGINE = InnoDB CHARACTER SET latin1 COLLATE latin1_spanish_ci;
